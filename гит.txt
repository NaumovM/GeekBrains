git log (origin/main ^main) - для просмотра лога изменений (поветочно)
git fetch - для затягиваний изменений без добавления.
git merge - для слияния.
git rebase - для слияния с сохранением коммитов.
git cherry-pick - для добавления определенного коммита, без всей ветки.
git clone path - для клонирования удаленного репозитария.
git show - чтобы показать последние изменения.
git diff <name> - посмотреть изменения.
git blame - посмотреть кто вносил изменения.
git stash pop - вытащить отложенные изменения.
-----------------------------------------------------------------
git restore (--staged) <file> - для отмены изменений незакомиченных (для файлов в индексе).
git reset --hard - удаляет все незакомиченные изменения в том числе трекед.
git clean -f - для удаления антрек файлов.
git rm --cached <name> - для удаления из отслеживания гит.
git revert <commit> - отмена конкретного коммита. (--no-commit)
git reset --soft <commit> - убирает коммит и оставляет его в индексе.
git reset --mixed <commit> - убирает коммит и убирает его из индекса.
git commit --amend -m<new> - для изменения текста последнегоо коммита.
git commit --amend --no-edit - для добавления в окммит новых изменений.
git reset --merge <commit> - отмена мерджа до предидущего коммита.
git merge --abort - отмена мерджа с конфликтом.
-----------------------------------------------------------------
git remote -v - для просмотра связей с удаленными репозитариями.
git remote remove origin - удалить связь с текущим репозитарием.
git remote add origin path -  добавить путь на удаленный репозитарий.
